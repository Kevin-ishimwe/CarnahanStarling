\chapter{Carnahan-Starling Equation of State}

\section{Introduction}

We formalize the Carnahan-Starling hard-sphere reference term for the residual Helmholtz free energy in the Statistical Associating Fluid Theory (SAFT) framework. The formalization derives from the compressibility factor through integration and algebraic manipulation, providing a machine-verified thermodynamic result.

\section{Physical Parameters}

\begin{definition}[Carnahan-Starling Primitive]
\label{def:primitive_vars}
\lean{CarnahanStarlingPrimitive}
\leanok
A Carnahan-Starling system is characterized by nine bundled physical quantities with validity constraints:
\begin{itemize}
  \item Compressibility factor $Z > 0$
  \item Segment diameter $d > 0$
  \item Number of particles $N \geq 1$
  \item Mathematical constant $\pi$
  \item Chain length $m \geq 1$
  \item Boltzmann constant $k_B > 0$
  \item Temperature $T \geq 0$
  \item Packing density $0 \leq \eta < 1$
  \item Number density $\rho \geq 0$
\end{itemize}
\end{definition}

\begin{theorem}
\label{thm:N_pos}
\lean{CarnahanStarlingPrimitive.N_pos}
\uses{def:primitive_vars}
\leanok
The number of particles is positive: $0 < p.N$.
\end{theorem}

\begin{theorem}
\label{thm:m_pos}
\lean{CarnahanStarlingPrimitive.m_pos}
\uses{def:primitive_vars}

\end{theorem}

\section{Compressibility Factor}

\begin{definition}[Carnahan-Starling Compressibility Factor]
\label{def:cs_compressibility_factor}
\lean{CarnahanStarlingPrimitive.csCompressibilityFactor}
\uses{def:primitive_vars}
\leanok
The Carnahan-Starling compressibility factor for hard-sphere fluids is:
$$Z^{\text{CS}}(\eta) = \frac{1 + \eta + \eta^2 - \eta^3}{(1 - \eta)^3}$$
\end{definition}

\begin{theorem}
\label{thm:cs_compressibility_factor_pos}
\lean{CarnahanStarlingPrimitive.csCompressibilityFactor_pos}
\uses{def:cs_compressibility_factor}
\leanok
For $\eta \in [0,1)$, the compressibility factor is positive: $0 < Z^{\text{CS}}(\eta)$.
\end{theorem}

\begin{theorem}
\label{thm:cs_compressibility_factor_at_zero}
\lean{CarnahanStarlingPrimitive.csCompressibilityFactor_at_zero}
\uses{def:cs_compressibility_factor}
\leanok
At $\eta = 0$ (ideal gas limit): $Z^{\text{CS}}(0) = 1$.
\end{theorem}

\section{Residual Helmholtz Free Energy}

\begin{definition}[Helmholtz Integral]
\label{def:helmholtz_from_z}
\lean{CarnahanStarlingPrimitive.helmholtzFromZ}
\uses{def:primitive_vars}
\leanok
The fundamental thermodynamic relation for residual Helmholtz free energy is:
$$\frac{A^{\text{res}}(ref)}{Nk_BT} = \int_0^\rho \frac{Z(\rho') - 1}{\rho'} d\rho'$$
\end{definition}

\section{Substitution and Simplification}

\begin{theorem}[Substitute Z into Integral]
\label{thm:substitute_z_into_integral}
\lean{SubstituteZIntoIntegral.substituteZIntoIntegral}
\uses{def:cs_compressibility_factor, def:helmholtz_from_z}
\leanok
Substituting the explicit Carnahan-Starling formula gives:
$$\frac{A^{\text{res}}(ref)}{Nk_BT} = \int_0^\rho \frac{\frac{1 + \eta(\rho') + \eta(\rho')^2 - \eta(\rho')^3}{(1 - \eta(\rho'))^3} - 1}{\rho'} d\rho'$$
\end{theorem}

\begin{theorem}[Simplify Integrand]
\label{thm:cs_numerator_identity}
\lean{CSSimplifyIntegrand.cs_numerator_identity}
\uses{def:cs_compressibility_factor}
\leanok
The integrand numerator simplifies to:
$$(1 + \eta + \eta^2 - \eta^3) - (1 - \eta)^3 = 4\eta - 2\eta^2$$
\end{theorem}

\begin{theorem}
\label{thm:cs_simplify_integrand}
\lean{CSSimplifyIntegrand.csSimplifyIntegrand}
\uses{thm:cs_numerator_identity, thm:substitute_z_into_integral}
\leanok
The simplified integrand is:
$$Z^{\text{CS}}(\eta) - 1 = \frac{4\eta - 2\eta^2}{(1 - \eta)^3}$$
\end{theorem}

\section{Change of Variables}

\begin{theorem}[Proportionality Integral Change]
\label{thm:proportionality_integral_change_of_var}
\lean{ProportionalityIntegralChangeOfVar.proportionalityIntegralChangeOfVar}
\uses{def:primitive_vars, def:packing_density_fn}
\leanok
Since $\eta = C \rho$ for constant $C = \frac{\pi}{6} m d^3 > 0$, integrals of functions divided by the variable are invariant:
$$\int_0^\rho \frac{g(C \rho')}{\rho'} d\rho' = \int_0^\eta \frac{g(\eta')}{\eta'} d\eta'$$
\end{theorem}

\begin{theorem}[Change Variable to $\eta$]
\label{thm:cs_change_variable_to_eta}
\lean{CSChangeVariableToEta.csChangeVariableToEta}
\uses{thm:cs_simplify_integrand, thm:proportionality_integral_change_of_var}
\leanok
Rewriting the Helmholtz integral entirely in terms of $\eta$:
$$\frac{A^{\text{res}}(ref)}{Nk_BT} = \int_0^\eta \frac{4\eta' - 2\eta'^2}{(1 - \eta')^3 \cdot \eta'} d\eta'$$
\end{theorem}

\section{Integrand Cancellation}

\begin{theorem}[Cancel $\eta$ in Integrand]
\label{thm:cs_integrand_cancel_eta}
\lean{ChangeVariableToEta.csIntegrandCancelEta}
\uses{thm:cs_change_variable_to_eta}
\leanok
Factoring and canceling $\eta$:
$$\frac{4\eta' - 2\eta'^2}{(1 - \eta')^3 \cdot \eta'} = \frac{4 - 2\eta'}{(1 - \eta')^3}$$
\end{theorem}

\section{Integral Evaluation via FTC}

\begin{definition}[CS Antiderivative]
\label{def:cs_antideriv}
\lean{SimplifyIntegrandEta.cs_antideriv}
\uses{thm:cs_integrand_cancel_eta}
\leanok
The antiderivative of $\frac{4 - 2\eta}{(1 - \eta)^3}$ is:
$$F(\eta) = \frac{4\eta - 3\eta^2}{(1 - \eta)^2}$$
\end{definition}

\begin{definition}[Carnahan-Starling Reference Term]
\label{def:cs_reference_term}
\lean{SimplifyIntegrandEta.csReferenceTerm}
\uses{def:cs_antideriv}
\leanok
The Carnahan-Starling hard-sphere reference term is:
$$\frac{A^{\text{res}}(ref)}{Nk_BT} = \frac{4\eta - 3\eta^2}{(1 - \eta)^2}$$
\end{definition}

\begin{theorem}
\label{thm:cs_reference_integral_eq}
\lean{SimplifyIntegrandEta.csReferenceIntegralEq}
\uses{def:cs_antideriv, def:cs_reference_term}
\leanok
By the Fundamental Theorem of Calculus:
$$\int_0^\eta \frac{4 - 2\eta'}{(1 - \eta')^3} d\eta' = \frac{4\eta - 3\eta^2}{(1 - \eta)^2}$$
\end{theorem}

\section{Substituting Back}

\begin{theorem}[Substitute Packing Density Back]
\label{thm:substitute_packing_density}
\lean{CSSubstitutePackingDensity.substitutePackingDensity}
\uses{def:cs_reference_term, def:packing_density_fn}
\leanok
Substituting $\eta = \frac{\pi}{6}\rho md^3$ into the reference term:
$$\frac{A^{\text{res}}(ref)}{Nk_BT} = \frac{4 \cdot \frac{\pi}{6}\rho md^3 - 3 \left(\frac{\pi}{6}\rho md^3\right)^2}{\left(1 - \frac{\pi}{6}\rho md^3\right)^2}$$
\end{theorem}

\section{Final Expanded Form}

\begin{definition}[Expanded CS Reference Term]
\label{def:expanded_cs_reference_term}
\lean{ExpandedCSReferenceTerm.expandedCSReferenceTerm}
\uses{def:cs_reference_term, def:packing_density_fn}
\leanok
Expanding the numerator coefficients:
$$\frac{A^{\text{res}}(ref)}{Nk_BT} = \frac{\frac{2\pi}{3}\rho md^3 - \frac{\pi^2}{12}\rho^2 m^2 d^6}{\left(1 - \frac{\pi}{6}\rho md^3\right)^2}$$
\end{definition}

\begin{theorem}
\label{thm:four_eta_expansion}
\lean{ExpandedCSReferenceTerm.four_eta_expansion}
\uses{def:packing_density_fn}
\leanok
Expansion of the first term:
$$4 \cdot \frac{\pi}{6}\rho md^3 = \frac{2\pi}{3}\rho md^3$$
\end{theorem}

\begin{theorem}
\label{thm:three_eta_sq_expansion}
\lean{ExpandedCSReferenceTerm.three_eta_sq_expansion}
\uses{def:packing_density_fn}
\leanok
Expansion of the second term:
$$3 \left(\frac{\pi}{6}\rho md^3\right)^2 = \frac{\pi^2}{12}\rho^2 m^2 d^6$$
\end{theorem}

\begin{theorem}[Final Form Equivalence]
\label{thm:expanded_cs_reference_term_eq_compact}
\lean{ExpandedCSReferenceTerm.expandedCSReferenceTerm_eq_compact}
\uses{def:expanded_cs_reference_term, def:cs_reference_term}
\leanok
The expanded form is equivalent to the compact packing-fraction form:
$$\frac{A^{\text{res}}(ref)}{Nk_BT} = \frac{4\eta - 3\eta^2}{(1 - \eta)^2}$$
\end{theorem}
